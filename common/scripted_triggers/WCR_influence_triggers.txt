# Westernization influence triggers

WCR_ruler_has_american_influence_trigger = {
    scope:character ?= {
        owner ?= {
            has_variable = WCR_ruler_influence
            THIS.var:WCR_ruler_influence = 1
        }
        home_country ?= {
            has_variable = WCR_ruler_influence
            THIS.var:WCR_ruler_influence = 1
        }
    }
}
