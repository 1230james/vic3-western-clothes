WCR_hairstyles = {
    usage = game
    selection_behavior = weighted_random

    WCR_westernized_asian_hair_styles = {
        dna_modifiers = {
            accessory = {
                mode = replace
                gene = hairstyles
                template = east_asian_hairstyles
                range = { 0 1 }
            }
        }
        weight = {
            base = 0

            # Purpose of this is to REMOVE the Japanese hairstyles when they westernize
            modifier = {
                add = 10000
                md_japanese_uniform_pack_active_trigger = no # JUP compatibility/deconflict
                WCR_st_completed_westernization_japanese_empire = yes
            }

            # Remove Empress Myeongseong hair
            modifier = {
                add = 10000
                WCR_st_completed_westernization_korean_empire = yes
                scope:character ?= {
                    has_template = KOR_empress_myeongseong
                }
                monarchy_clothes_trigger = no # See `WCR_no_korean_queen_hairstyle` for monarchy-specific version
            }
        }
    }

    # This one is for preserving unwesternized styles
    WCR_japanese_hair_styles = {
        dna_modifiers = {
            accessory = {
                mode = replace
                gene = hairstyles
                template = japanese_hairstyles
                range = { 0 1 } # For the randomness to work correctly
            }
        }
        weight = {
            base = 0
            modifier = {
                add = 1
                OR = {
                    AND = {
                        scope:pop ?= {
                            culture = cu:japanese
                        }
                        poor_clothes_pop_trigger = no
                    }
                    scope:character ?= {
                        culture = cu:japanese
                    }
                }
                WCR_st_completed_westernization_japanese_empire = no
            }
        }
    }

    # Remove Empress Myeongseong hair from westernized Korean queens (or queen dowagers); monarchy only
    WCR_no_korean_queen_hairstyle = {
        dna_modifiers = {
            accessory = {
                mode = replace
                gene = hairstyles
                template = east_asian_hairstyles
                range = { 0 0.5 } # [0.5, 1] range mixed w/ `european_royal_headgear` causes unwanted use of drills hairstyle
            }
        }
        weight = {
            base = 0
            modifier = {
                add = 10000
                WCR_st_completed_westernization_korean_empire = yes
                monarchy_clothes_trigger = yes
                OR = {
                    scope:character ?= {
                        has_template = KOR_empress_myeongseong
                    }
                    AND = { # Match w/ vanilla def for `korean_queen_hairstyle`
                        scope:character ?= {
                            culture = {
                                has_discrimination_trait = heritage_korean
                            }
                            is_female = yes
                            owner ?= {
                                country_has_primary_culture = cu:korean
                            }
                        }
                        # monarchy_clothes_trigger = yes
                    }
                }
            }
        }
    }
}
